# @zhangli2008/ai-editor-components

Business component library for the Ai editor platform.

## Usage

```tsx
import { LText } from '@zhangli2008/ai-editor-components'

export function Demo() {
  return (
    <LText
      text="Hello editor"
      fontSize={20}
      fontWeight="700"
      style={{ letterSpacing: 1 }}
    />
  )
}
```
